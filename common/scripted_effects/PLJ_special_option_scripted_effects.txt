PLJ_nuclear_reactor_2x = {

	random_owned_controlled_state = {
		limit = {
			free_building_slots = {
				building = nuclear_reactor
				size > 0
				include_locked = no
			}
		}
		add_building_construction = {
			type = nuclear_reactor
			level = 1
			instant_build = no #ADDS TO QUEUE
		}
